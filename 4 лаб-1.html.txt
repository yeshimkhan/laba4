
<script>
a= Number(prompt("A number :"," "));
b= Number(prompt("B number :"," "));
c= Number(prompt("C number :"," "));
d= Number(prompt("D number :"," "));
var arr = [a,b,c,d];
var min = arr[0];
function san(){
for (var k = 1; k < arr.length; k++) {
  if (arr[k] < min) {
    min = arr[k];
  }
}
return min;
}
console.log("a =" +a)
console.log("b =" +b)
console.log("c =" +c)
console.log("d =" +d)
console.log("Min =" +san());
</script>
